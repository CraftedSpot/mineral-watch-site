        // Completion report functions (loadCompletionReports, renderCompletionReport,

        //  are provided by shared-completion-reports.txt

        function toggleOccFilings(type) {
            const content = document.getElementById(type === 'property' ? 'property-occ-content' : 'well-occ-content');
            const arrow = document.getElementById(type === 'property' ? 'property-occ-arrow' : 'well-occ-arrow');
            if (content.style.display === 'none') {
                content.style.display = 'block';
                arrow.style.transform = 'rotate(90deg)';
            } else {
                content.style.display = 'none';
                arrow.style.transform = 'rotate(0deg)';
            }
        }
        window.toggleOccFilings = toggleOccFilings;

        // Thin wrapper: calls shared loadCompletionReports with dashboard-specific element IDs
        function loadWellRecords(apiNumber) {

                countBadge: document.getElementById('well-records-count')

        }
        window.loadWellRecords = loadWellRecords;

        // Thin wrapper: calls shared loadOccFilingsShared with dashboard-specific element IDs
        function loadOccFilings(section, township, range, type = 'property', meridian = 'IM', apiNumber = null) {
            const prefix = type === 'property' ? 'property' : 'well';
            loadOccFilingsShared(
                { section, township, range, meridian, type, apiNumber },
                {
                    directContainer: document.getElementById(`${prefix}-occ-direct`),
                    adjacentContainer: document.getElementById(`${prefix}-occ-adjacent`),
                    adjacentWrapper: document.getElementById(`${prefix}-occ-adjacent-wrapper`),
                    emptyEl: document.getElementById(`${prefix}-occ-empty`),
                    countBadge: document.getElementById(`${prefix}-occ-count`),
                    adjacentCountBadge: document.getElementById(`${prefix}-occ-adjacent-count`)
                }
            );
        }
        window.loadOccFilings = loadOccFilings;
