// Format property label in TRS format: County — 19N-25W-09
function formatPropertyLabel(p) {
  var trs = formatTRS(p.township, p.range, p.section);
  return (p.county || '?') + ' — ' + trs;
}

async function loadTitlePage() {
  var timeline = document.getElementById('titleTimeline');
  timeline.innerHTML = '<div class="title-loading"><div class="spinner"></div>Loading title chain data...</div>';

  try {
    var resp = await fetch('/api/title-chain/properties');
    if (!resp.ok) throw new Error('Failed to load');
    var data = await resp.json();

    titleProperties = data.properties || [];

    if (titleProperties.length === 0 && (data.unlinkedCount || 0) === 0) {
      timeline.innerHTML = '<div class="title-empty">' +
        '<svg width="48" height="48" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="1.5" d="M12 8v4l3 3m6-3a9 9 0 11-18 0 9 9 0 0118 0z"/></svg>' +
        '<p>No chain-of-title documents found. Upload mineral deeds, leases, assignments, and other title documents to build your chain of title.</p></div>';
      return;
    }

    renderPropertySelector(data.unlinkedCount || 0);

    // Check for ?property= query param
    var params = new URLSearchParams(window.location.search);
    var propertyParam = params.get('property');
    if (propertyParam) {
      var bareId = propertyParam.startsWith('prop_') ? propertyParam.slice(5) : propertyParam;
      var match = titleProperties.find(function(p) { return p.airtableRecordId === bareId; });
      if (match) {
        selectTitleProperty(match.airtableRecordId);
        return;
      }
    }

    // Default: select first property
    if (titleProperties.length > 0) {
      selectTitleProperty(titleProperties[0].airtableRecordId);
    }
  } catch (err) {
    console.error('[Title] Load error:', err);
    timeline.innerHTML = '<div class="title-empty"><p>Failed to load title chain data. Please try again.</p></div>';
  }
}

function renderPropertySelector(unlinkedCount) {
  var area = document.getElementById('titleSelectorArea');

  var html = '<button class="title-property-btn" onclick="toggleTitleDropdown()" id="titlePropertyBtn">' +
    '<span id="titlePropertyLabel">Select a property...</span>' +
    '<svg width="16" height="16" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 9l-7 7-7-7"/></svg>' +
    '</button>' +
    '<div class="title-property-dropdown" id="titleDropdown">' +
    '<input type="text" class="title-property-search" id="titlePropertySearch" placeholder="Filter by county or legal..." oninput="filterTitleProperties()">' +
    '<div class="title-property-list" id="titlePropertyList"></div>' +
    '</div>';

  area.innerHTML = html;
  renderTitlePropertyList('');

  // Unlinked section in timeline area
  if (unlinkedCount > 0) {
    var timeline = document.getElementById('titleTimeline');
    var unlinkedHtml = '<div class="title-unlinked" id="titleUnlinkedSection">' +
      '<button class="title-unlinked-toggle" onclick="toggleTitleUnlinked()">' +
      '<svg width="12" height="12" fill="none" stroke="currentColor" viewBox="0 0 24 24" id="titleUnlinkedArrow" style="transition:transform 0.15s"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"/></svg>' +
      unlinkedCount + ' unlinked chain document' + (unlinkedCount !== 1 ? 's' : '') +
      '</button>' +
      '<div class="title-unlinked-list" id="titleUnlinkedList"></div>' +
      '</div>';
    // We'll append it after timeline renders
    window._titleUnlinkedHtml = unlinkedHtml;
    window._titleUnlinkedCount = unlinkedCount;
  }
}

function renderTitlePropertyList(filter) {
  var list = document.getElementById('titlePropertyList');
  if (!list) return;
  var lower = filter.toLowerCase().trim();
  var filtered = titleProperties.filter(function(p) {
    if (!lower) return true;
    var trs = formatTRS(p.township, p.range, p.section).toLowerCase();
    var county = (p.county || '').toLowerCase();
    var full = county + ' ' + trs;
    // Strip leading zeros from search term for flexible matching (9 matches 09)
    var stripped = lower.replace(/\b0+(\d)/g, '$1');
    var trsStripped = trs.replace(/\b0+(\d)/g, '$1');
    return full.indexOf(lower) >= 0 || full.indexOf(stripped) >= 0
      || trs.indexOf(lower) >= 0 || trsStripped.indexOf(stripped) >= 0;
  });

  list.innerHTML = filtered.map(function(p) {
    var label = formatPropertyLabel(p);
    var sel = titleSelectedPropertyId === p.airtableRecordId ? ' selected' : '';
    return '<div class="title-property-option' + sel + '" onclick="selectTitleProperty(\'' + p.airtableRecordId + '\')">' +
      '<span>' + escapeHtml(label) + '</span>' +
      '<span class="doc-count">' + p.chainDocCount + ' doc' + (p.chainDocCount !== 1 ? 's' : '') + '</span></div>';
  }).join('');
}

function toggleTitleDropdown() {
  titleDropdownOpen = !titleDropdownOpen;
  var dd = document.getElementById('titleDropdown');
  var btn = document.getElementById('titlePropertyBtn');
  if (titleDropdownOpen) {
    dd.classList.add('visible');
    btn.classList.add('open');
    var search = document.getElementById('titlePropertySearch');
    if (search) { search.value = ''; search.focus(); }
    renderTitlePropertyList('');
  } else {
    dd.classList.remove('visible');
    btn.classList.remove('open');
  }
}

function filterTitleProperties() {
  var search = document.getElementById('titlePropertySearch');
  renderTitlePropertyList(search ? search.value : '');
}

// Close dropdown on outside click

async function selectTitleProperty(airtableRecordId) {
  titleSelectedPropertyId = airtableRecordId;

  // Close dropdown
  titleDropdownOpen = false;
  var dd = document.getElementById('titleDropdown');
  var btn = document.getElementById('titlePropertyBtn');
  if (dd) dd.classList.remove('visible');
  if (btn) btn.classList.remove('open');

  // Update button label
  var prop = titleProperties.find(function(p) { return p.airtableRecordId === airtableRecordId; });
  if (prop) {
    document.getElementById('titlePropertyLabel').textContent = formatPropertyLabel(prop);
  }
  renderTitlePropertyList('');

  // Load chain
  var timeline = document.getElementById('titleTimeline');
  var stats = document.getElementById('titleStats');
  timeline.innerHTML = '<div class="title-loading"><div class="spinner"></div>Loading chain of title...</div>';
  stats.innerHTML = '';

  try {
    var resp = await fetch('/api/property/' + encodeURIComponent(airtableRecordId) + '/title-chain?include_unlinked=1');
    if (!resp.ok) throw new Error('Failed to load chain');
    titleChainData = await resp.json();

    renderTitleTimeline(titleChainData);
    renderTitleStats(titleChainData);

    // Append unlinked section if exists
    if (window._titleUnlinkedCount > 0) {
      var existing = document.getElementById('titleUnlinkedSection');
      if (!existing) {
        timeline.insertAdjacentHTML('beforeend', window._titleUnlinkedHtml);
      }
      renderTitleUnlinkedDocs(titleChainData.unlinkedDocuments || []);
    }
  } catch (err) {
    console.error('[Title] Chain load error:', err);
    timeline.innerHTML = '<div class="title-empty"><p>Failed to load chain of title for this property.</p></div>';
  }
}

function renderTitleStats(data) {
  var statsEl = document.getElementById('titleStats');
  if (!statsEl) return;

  var docs = data.documents || [];
  if (docs.length === 0) { statsEl.innerHTML = ''; return; }

  var count = docs.length;
  var dates = docs.map(function(d) { return d.date; }).filter(Boolean).sort();
  var dateRange = dates.length > 0 ? (dates[0].substring(0, 4) + '–' + dates[dates.length - 1].substring(0, 4)) : '';

  // Count unique parties
  var partySet = {};
  docs.forEach(function(d) {
    (d.parties || []).forEach(function(p) {
      partySet[p.name.toLowerCase()] = true;
    });
  });
  var partyCount = Object.keys(partySet).length;

  var parts = [count + ' document' + (count !== 1 ? 's' : '')];
  if (dateRange) parts.push(dateRange);
  if (partyCount > 0) parts.push(partyCount + ' part' + (partyCount !== 1 ? 'ies' : 'y'));

  statsEl.innerHTML = parts.join('<span class="stats-dot">&middot;</span>');
}

function renderTitleParties(doc) {
  var parties = doc.parties || [];
  if (parties.length === 0) return '';

  var html = '<div class="title-parties">';
  var arrowRoles = TITLE_ARROW_TYPES[doc.docType];

  if (arrowRoles) {
    var fromRole = arrowRoles[0];
    var toRole = arrowRoles[1];
    var froms = parties.filter(function(p) { return p.role === fromRole; });
    var tos = parties.filter(function(p) { return p.role === toRole; });

    if (froms.length > 0 && tos.length > 0) {
      html += escapeHtml(froms.map(function(p) { return p.name; }).join(', '));
      html += '<span class="arrow">&rarr;</span>';
      html += escapeHtml(tos.map(function(p) { return p.name; }).join(', '));
    } else {
      html += parties.map(function(p) { return escapeHtml(p.name); }).join(', ');
    }
  } else if (doc.docType === 'affidavit_of_heirship' || doc.docType === 'probate') {
    var decedent = parties.find(function(p) { return p.role === 'grantor' || p.role === 'decedent'; });
    var heirs = parties.filter(function(p) { return p.role === 'grantee'; });
    if (decedent) {
      html += 'Estate of ' + escapeHtml(decedent.name);
      if (heirs.length > 0) {
        html += '<div class="party-list">';
        heirs.forEach(function(h) {
          html += '<span class="party-list-item">&#8627; ' + escapeHtml(h.name) + '</span>';
        });
        html += '</div>';
      }
    } else {
      html += parties.map(function(p) { return escapeHtml(p.name); }).join(', ');
    }
  } else if (doc.docType === 'title_opinion') {
    var examiner = parties.find(function(p) { return p.role === 'examiner'; });
    var owners = parties.filter(function(p) { return p.role === 'owner'; });
    if (examiner) {
      html += '<span class="party-label">Examined by</span>' + escapeHtml(examiner.name);
    }
    if (owners.length > 0) {
      html += '<div class="party-list">';
      owners.forEach(function(o) {
        html += '<span class="party-list-item">' + escapeHtml(o.name) + '</span>';
      });
      html += '</div>';
    }
    if (!examiner && owners.length === 0) {
      html += parties.map(function(p) { return escapeHtml(p.name); }).join(', ');
    }
  } else if (doc.docType === 'death_certificate') {
    var dec = parties.find(function(p) { return p.role === 'decedent'; }) || parties[0];
    if (dec) html += escapeHtml(dec.name);
  } else {
    html += parties.map(function(p) { return escapeHtml(p.name); }).join(', ');
  }

  html += '</div>';
  return html;
}

function formatTitleDate(dateStr) {
  if (!dateStr) return '';
  try {
    var parts = dateStr.split('-');
    if (parts.length === 3) {
      var months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
      var m = parseInt(parts[1], 10) - 1;
      var d = parseInt(parts[2], 10);
      return months[m] + ' ' + d + ', ' + parts[0];
    }
    return dateStr;
  } catch (e) { return dateStr; }
}

      '<strong>' + escapeHtml(doc.displayName || 'Untitled') + '</strong>' +
      (dateStr ? ' &middot; ' + dateStr : '') +
      '</div>';
  }).join('');
}

// ─── Duplicate Comparison Modal ───
var _dupBlobUrls = [];
var _dupCurrentCandidateId = null;

function openDupCompare(candidateId, keeperId, matchType, candidateName, keeperName) {
  _dupCurrentCandidateId = candidateId;
  var modal = document.getElementById('dupCompareModal');
  if (!modal) return;
  modal.style.display = 'flex';

  // Set header names
  document.getElementById('dupKeeperName').textContent = keeperName || 'Keeper';
  document.getElementById('dupCandidateName').textContent = candidateName || 'Candidate';

  // Match description
  var desc = 'These documents ';
  if (matchType === 'instrument_number') desc += 'share the same instrument number and county.';
  else if (matchType === 'book_page') desc += 'share the same book and page in the same county.';
  else desc += 'have matching parties, date, and legal description.';
  document.getElementById('dupMatchDesc').textContent = desc;

  // Load PDFs
  _loadDupPane(keeperId, 'dupKeeperPane');
  _loadDupPane(candidateId, 'dupCandidatePane');
}

function _loadDupPane(docId, containerId) {
  var container = document.getElementById(containerId);
  if (!container) return;
  container.innerHTML = '<div style="display: flex; align-items: center; justify-content: center; height: 100%; color: white; font-size: 14px;">Loading PDF...</div>';

  fetch('/api/documents/' + docId + '/download?view=true', { credentials: 'include' })
    .then(function(resp) {
      if (!resp.ok) throw new Error('Failed to fetch PDF');
      return resp.blob();
    })
    .then(function(blob) {
      var url = URL.createObjectURL(blob);
      _dupBlobUrls.push(url);
      container.innerHTML = '<iframe src="' + url + '#view=FitH" style="width: 100%; height: 100%; border: none;"></iframe>';
    })
    .catch(function(err) {
      container.innerHTML = '<div style="display: flex; align-items: center; justify-content: center; height: 100%; color: #FCA5A5; font-size: 14px;">Failed to load PDF</div>';
      console.error('[DupCompare] Load error:', err);
    });
}

function confirmDuplicate() {
  if (!_dupCurrentCandidateId) return;
  fetch('/api/documents/' + _dupCurrentCandidateId + '/duplicate-review', {
    method: 'POST',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ action: 'confirm' })
  }).then(function(resp) {
    if (!resp.ok) throw new Error('Failed');
    closeDupCompare();
    selectTitleProperty(titleSelectedPropertyId);
  }).catch(function(err) {
    alert('Failed to confirm duplicate. Please try again.');
    console.error('[DupCompare] Confirm error:', err);
  });
}

function dismissDuplicate() {
  if (!_dupCurrentCandidateId) return;
  fetch('/api/documents/' + _dupCurrentCandidateId + '/duplicate-review', {
    method: 'POST',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ action: 'dismiss' })
  }).then(function(resp) {
    if (!resp.ok) throw new Error('Failed');
    closeDupCompare();
    selectTitleProperty(titleSelectedPropertyId);
  }).catch(function(err) {
    alert('Failed to dismiss duplicate. Please try again.');
    console.error('[DupCompare] Dismiss error:', err);
  });
}

function closeDupCompare() {
  var modal = document.getElementById('dupCompareModal');
  if (modal) modal.style.display = 'none';
  // Clean up blob URLs
  _dupBlobUrls.forEach(function(url) { try { URL.revokeObjectURL(url); } catch(e) {} });
  _dupBlobUrls = [];
  _dupCurrentCandidateId = null;
  // Clear panes
  var k = document.getElementById('dupKeeperPane');
  var c = document.getElementById('dupCandidatePane');
  if (k) k.innerHTML = '';
  if (c) c.innerHTML = '';
}

function confirmAllDuplicates() {
  var dups = (titleChainData && titleChainData.pendingDuplicates) || [];
  if (dups.length === 0) return;
  if (!confirm('Confirm all ' + dups.length + ' document' + (dups.length > 1 ? 's' : '') + ' as duplicates? They will be hidden from the title chain.')) return;

  var ids = dups.map(function(d) { return d.id; });
  fetch('/api/documents/duplicate-review-batch', {
    method: 'POST',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ action: 'confirm', document_ids: ids })
  }).then(function(resp) {
    if (!resp.ok) throw new Error('Failed');
    selectTitleProperty(titleSelectedPropertyId);
  }).catch(function(err) {
    alert('Failed to confirm duplicates. Please try again.');
    console.error('[DupCompare] Batch confirm error:', err);
  });
}
